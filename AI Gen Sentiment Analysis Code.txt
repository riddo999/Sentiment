# Import necessary libraries
import pandas as pd
from transformers import pipeline
import matplotlib.pyplot as plt

# Load the dataset
df = pd.read_csv('book_reviews_sample.csv')

# Check if 'reviewText' column exists
if 'reviewText' not in df.columns:
    print("Column 'reviewText' not found in the CSV file.")
    print("Available columns:", df.columns.tolist())
else:
    # Initialize the sentiment analysis pipeline
    sentiment_analyzer = pipeline("sentiment-analysis")
    
    # Function to analyze sentiment (with handling for long texts)
    def analyze_sentiment(text):
        if pd.isna(text):
            return {"label": "NEUTRAL", "score": 0.5}
        
        # Truncate long texts to avoid model limitations
        if len(text) > 500:
            text = text[:500]
        
        try:
            result = sentiment_analyzer(text)[0]
            return result
        except Exception as e:
            print(f"Error analyzing text: {e}")
            return {"label": "ERROR", "score": 0.0}
    
    # Apply sentiment analysis to each review
    print("Analyzing sentiments...")
    df['sentiment'] = df['reviewText'].apply(lambda x: analyze_sentiment(x))
    
    # Extract label and score
    df['sentiment_label'] = df['sentiment'].apply(lambda x: x['label'])
    df['sentiment_score'] = df['sentiment'].apply(lambda x: x['score'])
    
    # Display results
    print("\nSentiment Distribution:")
    sentiment_counts = df['sentiment_label'].value_counts()
    print(sentiment_counts)
	    # Visualize the sentiment distribution
    plt.figure(figsize=(10, 6))
    sentiment_counts.plot(kind='bar', color=['red', 'green'] if len(sentiment_counts) == 2 else 'blue')
    plt.title('Sentiment Distribution in Book Reviews')
    plt.xlabel('Sentiment')
    plt.ylabel('Count')
    plt.xticks(rotation=0)
    plt.tight_layout()
    plt.show()
    
    # Display some sample reviews with their sentiment
    print("\nSample Reviews with Sentiment Analysis:")
    sample_df = df[['reviewText', 'sentiment_label', 'sentiment_score']].head(5)
    for i, row in sample_df.iterrows():
        print(f"\nReview {i+1}:")
        print(f"Text: {row['reviewText'][:100]}..." if len(str(row['reviewText'])) > 100 else f"Text: {row['reviewText']}")
        print(f"Sentiment: {row['sentiment_label']} (Score: {row['sentiment_score']:.4f})")
    
    # Save the results to a new CSV file
    df.to_csv('book_reviews_with_sentiment.csv', index=False)
    print("\nResults saved to 'book_reviews_with_sentiment.csv'")
